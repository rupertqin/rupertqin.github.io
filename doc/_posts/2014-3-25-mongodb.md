drop collection:

    db.collection.drop()

delete property, $unset prop to '', and the third param apply to all:

    db.collection.update({prop_a: true}, {$unset: {prop_b: ''} }, true)


## start

start server:

    mongod --config /usr/local/etc/mongod.conf

start shell:

    mongo
