## groups
`(0{1,})|(abs){2}|([1-9][0-9]*)`

email: `^([\w.-]+)@([\da-z\-]+)(\.([a-z]{2,6})){1,2}$`

URL: `^(https?:\/\/)?([\da-z\.-]+)\.([a-z\.]{2,6})([\/\w \.-]*)*\/?$`

IP: `^(?:(?:25[0-5]|2[0-4][0-9]|[01]?[0-9][0-9]?)\.){3}(?:25[0-5]|2[0-4][0-9]|[01]?[0-9][0-9]?)$`

HTML tag: `^<([a-z]+)([^<]+)*(?:>(.*)<\/\1>|\s+\/>)$`
